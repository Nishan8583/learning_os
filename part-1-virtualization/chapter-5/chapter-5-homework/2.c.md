Write a program that opens a file (with the open() system call) and then calls fork() to create a new process. Can both the child and parent access the file descriptor returned by open()? What happens when they are writing to the file concurrently, i.e., at the same time?
- They both can access the fd returned by the open()
```c
#include <stdio.h>
#include <fcntl.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

int main() {
    char *filename = "test.txt";

    int fd = open(filename, O_RDWR | O_CREAT,0666);
    if (fd == -1) {
        perror("Error opening file");
        exit(1);
    }
    printf("File '%s' opened successfully with file descriptor %d\n", filename, fd);

    int rc = fork();
    if (rc < 0) {
        perror("Fork failed");
        close(fd);
        exit(1);
    } else if (rc == 0) {
        // In child
        char *content = "Child is writing\n";
        int r = write(fd, content, strlen(content));
        if (r == -1) {
            perror("Child write failed");
            close(fd);
            exit(1);
        }
    } else {
        // In parent
        char *content = "Parent is writing\n";
        int r = write(fd, content, strlen(content));
        if (r == -1) {
            perror("Parent write failed");
            close(fd);
            exit(1);
        }
    }
    close(fd);
    return 1; 
}
```
